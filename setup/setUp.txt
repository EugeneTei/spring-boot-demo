# set up scylla-db container
docker-compose up -d

# enter container
docker exec -it scylla cqlsh

# create keyspace
CREATE KEYSPACE IF NOT EXISTS example WITH REPLICATION = {'class': 'SimpleStrategy', 'replication_factor' : 1};

# create
USE example;
CREATE TABLE IF NOT EXISTS chatroom (
            item_id int,
            user_source TEXT,
            user_id int,
            time TIMESTAMP,
            PRIMARY KEY ((user_source, user_id), item_id)
    );
SELECT * FROM chatroom;

# compare
CREATE TABLE IF NOT EXISTS compare (
            item_id int,
            user_source TEXT,
            user_id int,
            time TIMESTAMP,
            PRIMARY KEY ((user_source, user_id), item_id)
    );
